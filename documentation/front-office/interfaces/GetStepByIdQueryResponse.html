import {gql} from 'apollo-angular';

import { Dashboard, Application, Form, User, Workflow, Permission, Step, Page} from '@safe/builder';

// === GET USERS ===
export const GET_USERS = gql`
{
  users {
    id
    username
    name
    roles {
      id
      title
    }
    oid
  }
}`;

export interface GetUsersQueryResponse {
  loading: boolean;
  users: User[];
}
